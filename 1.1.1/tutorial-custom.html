        var uploader = new tui.component.Uploader({
        url: {
            send: 'http://localhost:3000/upload',
            remove: 'http://localhost:3000/remove'
        },
        redirectURL: 'http://localhost:8080/samples/result.html',
        isBatchTransfer: false,
        listInfo: {
            list: $('#files'),
            count: $('#file_count'),
            size: $('#size_count')
        },
        fileField: 'userfile[]',
        isMultiple: true
    }, $('#uploader'));

    uploader.on('success', function(data) {
        $('#status').html(data.success + ' files added');
    });
    uploader.on('remove', function(data) {
        $('#status').html(data.name + ' file removed');
    });
